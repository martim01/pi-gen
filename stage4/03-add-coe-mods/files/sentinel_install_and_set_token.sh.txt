#!/bin/sh

set -e

if [ `id -u` -ne 0 ]; then
    echo "sudo is required for this script to execute successfully."
    exit 1
fi

if [ ! -f "./sentinel.deb" ]; then
    echo "Could not install sentinel -- Please ensure the file has been copied and renamed to sentinel.deb"
    exit 1
fi

dpkg -i ./sentinel.deb
/opt/sentinelone/bin/sentinelctl management token set longTokenGoesHereDependingOnSite
/opt/sentinelone/bin/sentinelctl management proxy set http://185.46.212.98:9400
/opt/sentinelone/bin/sentinelctl control start
